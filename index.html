import fs from 'fs'; import path from 'path'; export interface Order { id: string; name: string; email: string; phone: string; business: string; description: string; offerType: 'essentielle' | 'premium'; amount: number; // en centimes d'euros status: 'pending' | 'succeeded' | 'failed'; paymentIntentId: string; createdAt: string; updatedAt: string; } export interface TransactionLog { id: string; timestamp: string; level: 'info' | 'warn' | 'error'; action: string; orderId?: string; paymentIntentId?: string; message: string; details?: any; } const DATA_DIR = path.join(process.cwd(), 'data'); const DB_FILE = path.join(DATA_DIR, 'database.json'); interface Schema { orders: Order[]; logs: TransactionLog[]; } function ensureDb() { if (!fs.existsSync(DATA_DIR)) { fs.mkdirSync(DATA_DIR, { recursive: true }); } if (!fs.existsSync(DB_FILE)) { const initialData: Schema = { orders: [ { id: 'ord_demo_1', name: 'Jean Dupont', email: 'jean.dupont@example.com', phone: '0612345678', business: 'Boulangerie Dupont', description: 'Site vitrine pour ma boulangerie artisanale.', offerType: 'essentielle', amount: 7000, status: 'succeeded', paymentIntentId: 'pi_demo_1', createdAt: new Date(Date.now() - 3600000 * 2).toISOString(), updatedAt: new Date(Date.now() - 3600000 * 2).toISOString(), } ], logs: [ { id: 'log_1', timestamp: new Date().toISOString(), level: 'info', action: 'SERVER_BOOT', message: 'Base de données et système de journalisation initialisés.' } ] }; fs.writeFileSync(DB_FILE, JSON.stringify(initialData, null, 2), 'utf-8'); } } export function readDb(): Schema { ensureDb(); try { const content = fs.readFileSync(DB_FILE, 'utf-8'); return JSON.parse(content); } catch (error) { console.error('Erreur de lecture de la base de données:', error); return { orders: [], logs: [] }; } } export function writeDb(data: Schema) { ensureDb(); try { fs.writeFileSync(DB_FILE, JSON.stringify(data, null, 2), 'utf-8'); } catch (error) { console.error('Erreur d\'écriture de la base de données:', error); } } export function addOrder(order: Omit): Order { const db = readDb(); const newOrder: Order = { ...order, id: 'ord_' + Math.random().toString(36).substr(2, 9), createdAt: new Date().toISOString(), updatedAt: new Date().toISOString() }; db.orders.unshift(newOrder); writeDb(db); return newOrder; } export function updateOrderStatusByPaymentIntent(paymentIntentId: string, status: 'succeeded' | 'failed'): Order | null { const db = readDb(); const orderIndex = db.orders.findIndex(o => o.paymentIntentId === paymentIntentId); if (orderIndex === -1) return null; db.orders[orderIndex].status = status; db.orders[orderIndex].updatedAt = new Date().toISOString(); const updatedOrder = db.orders[orderIndex]; writeDb(db); return updatedOrder; } export function addLog(log: Omit): TransactionLog { const db = readDb(); const newLog: TransactionLog = { ...log, id: 'log_' + Math.random().toString(36).substr(2, 9), timestamp: new Date().toISOString() }; db.logs.unshift(newLog); writeDb(db); return newLog; } export function getOrders(): Order[] { return readDb().orders; } export function getLogs(): TransactionLog[] { return readDb().logs; }
